```python
class BaseRepository(Generic[T]):
    def __init__(self, model: Type[T], db: Session) -> None:
        self._model = model
        self._db = db

    def create(self, instance: T) -> T:
        """Persist a new entity; flush + refresh so DB-generated fields populate."""
        self._db.add(instance)
        self._db.flush()
        self._db.refresh(instance)
        return instance

    def get_by_id(self, record_id: UUID) -> T | None:
        return self._db.get(self._model, record_id)
```

    ```python
    """
    ORM entity for Stage 3 backend generation run records.

    Each row represents one complete pipeline execution: from djcli invocation
    through code generation to the final PipelineOutput.  The row is written

    processing → completed) is handled by the Job entity.
    """

    from __future__ import annotations

    import uuid

    from sqlalchemy import Boolean, DateTime, ForeignKey, JSON, String, Text, func
    from sqlalchemy.orm import Mapped, mapped_column

    from app.core.config import settings
    from src.msbc.database.base import Base


    def _uuid_col():
        """Return a UUID column type compatible with both PostgreSQL and SQLite."""
        if settings.DATABASE_URL.startswith("sqlite"):
            return String(36)
        from sqlalchemy.dialects.postgresql import UUID
        return UUID(as_uuid=True)


    class BackendGeneration(Base):
        """
        Persists the result of one Stage 3 backend generation run.

        Table: backend_generations

        Columns (D-03 — exact set, no extras):
          id              — UUID PK
          extraction_id   — FK → requirement_extractions.id (not nullable, indexed)
          project_name    — snake_case djcli project name (String(255))
          output_path     — absolute disk path where Django project landed (String(1024))
          cli_stdout      — raw djcli stdout, nullable (debug aid)
          cli_stderr      — raw djcli stderr, nullable (debug aid)
          pipeline_output — full PipelineOutput.model_dump(), nullable JSON
          success         — Boolean; True only when all apps generated and syntax-valid
          created_at      — UTC timestamp, server_default=func.now()
        """

        __tablename__ = "backend_generations"

        id: Mapped[str] = mapped_column(
            _uuid_col(),
            primary_key=True,
            default=lambda: str(uuid.uuid4()),
            comment="Unique backend generation run identifier",
        )
        extraction_id: Mapped[str] = mapped_column(
            String(36),
            ForeignKey("requirement_extractions.id", ondelete="CASCADE"),
            nullable=False,
            index=True,
            comment="FK → requirement_extractions.id — the source extraction this generation was derived from",
        )
        project_name: Mapped[str] = mapped_column(
            String(255),
            nullable=False,
            comment="snake_case djcli project name",
        )
        output_path: Mapped[str] = mapped_column(
            String(1024),
            nullable=False,
            comment="Absolute disk path where the generated Django project landed",
        )
        cli_stdout: Mapped[str | None] = mapped_column(
            Text,
            nullable=True,
            comment="Raw djcli stdout — debug aid",
        )
        cli_stderr: Mapped[str | None] = mapped_column(
            Text,
            nullable=True,
            comment="Raw djcli stderr — debug aid",
        )
        pipeline_output: Mapped[dict | None] = mapped_column(
            JSON,
            nullable=True,
            comment="Full PipelineOutput.model_dump() — populated on success only",
        )
        success: Mapped[bool] = mapped_column(
            Boolean,
            nullable=False,
            comment="True if the pipeline completed without errors",
        )
        created_at: Mapped[str] = mapped_column(
            DateTime(timezone=True),
            nullable=False,
            server_default=func.now(),
            comment="UTC timestamp of row creation",
        )

        def __repr__(self) -> str:
            return (
                f"<BackendGeneration id={self.id!s} "
                f"extraction_id={self.extraction_id!r} "
                f"success={self.success!r}>"
            )
    ```
