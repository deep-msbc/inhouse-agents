    """
    CLI Invoker — djcli subprocess wrapper for Stage 3.

    Invokes the django_cli_tool package (run via `python -m djcli`) to scaffold
    a new Django DRF project with one app per extraction module.

    Locked invariants (CLAUDE.md §6):
      - --api is ALWAYS passed. Never omit it.
      - --auth is NEVER passed. Not now, not ever.
      - subprocess timeout is always 60 seconds.
      - The djcli path is resolved via `python -m djcli` (not a hardcoded .exe).
    """

    from __future__ import annotations

    import logging
    import os
    import subprocess
    from typing import NamedTuple

    from src.msbc.models.schemas.backend_pipeline import CLIInvokerInput, CLIInvokerOutput

    logger = logging.getLogger(__name__)


    class CLIInvokerResult(NamedTuple):
        """Wraps CLIInvokerOutput with raw subprocess streams for ORM persistence."""
        output: CLIInvokerOutput
        stdout: str
        stderr: str


    class CLIInvoker:
        """
        Wraps the `django_cli_tool` (djcli) subprocess call.

        Usage::

            invoker = CLIInvoker()
            result = invoker.invoke(cli_input)
            # result.output  -> CLIInvokerOutput (structured)
            # result.stdout  -> raw djcli stdout (for BackendGeneration row)
            # result.stderr  -> raw djcli stderr (for BackendGeneration row)
        """

        def invoke(self, input: CLIInvokerInput) -> CLIInvokerResult:
            """
            Execute djcli and return structured output + raw streams.

            Guarantees:
              - output_path directory is created (exist_ok=True) before djcli runs.
              - --api is always included; --auth is never included.
              - TimeoutExpired is caught and mapped to success=False.
            """
            # D-02: ensure the output directory exists before djcli runs
            if not os.path.isabs(input.output_path):
                raise ValueError(
                    f"output_path must be an absolute path, got: {input.output_path!r}"
                )
            os.makedirs(input.output_path, exist_ok=True)

            cmd = [
                "python", "-m", "djcli",
                input.command,          # "startproject" | "startapp" | "noop"
                input.project_name,
                *input.app_names,
                "--api",                # LOCKED: always present (CLAUDE.md §6 rule 9)
                "--path", input.output_path,
                # NEVER --auth (CLAUDE.md §6 rule 8)
            ]

            logger.info(
                "Invoking djcli: command=%s project=%s apps=%s output_path=%s",
                input.command,
                input.project_name,
                input.app_names,
                input.output_path,
            )

            try:
                result = subprocess.run(
                    cmd,
                    capture_output=True,
                    text=True,
                    timeout=60,
                )
            except subprocess.TimeoutExpired:
                logger.error("djcli timed out after 60s for project=%s", input.project_name)
                timeout_output = CLIInvokerOutput(
                    project_path=os.path.join(input.output_path, input.project_name),
                    framework=input.framework,
                    generated_apps=[],
                    success=False,
                    errors=["djcli timed out after 60s"],  # exact text per CLAUDE.md §6
                )
                return CLIInvokerResult(output=timeout_output, stdout="", stderr="")

            success = result.returncode == 0
            errors: list[str] = []
            if not success:
                stderr_msg = result.stderr.strip()
                errors.append(stderr_msg or "djcli exited non-zero")
                logger.error(
                    "djcli failed: returncode=%d stderr=%r",
                    result.returncode,
                    result.stderr,
                )
            else:
                logger.info(
                    "djcli succeeded: project=%s apps=%s",
                    input.project_name,
                    input.app_names,
                )

            cli_output = CLIInvokerOutput(
                project_path=os.path.join(input.output_path, input.project_name),
                framework=input.framework,
                generated_apps=input.app_names if success else [],
                success=success,
                errors=errors,
            )
            return CLIInvokerResult(
                output=cli_output,
                stdout=result.stdout,
                stderr=result.stderr,
            )
    ```

    """
    Scaffold Validator — verifies the djcli-generated file tree for Stage 3.

    After `CLIInvoker.invoke()` returns success, this validator confirms that
    every expected app file exists before code generation begins.  A missing
    file means djcli did not scaffold the app correctly and code generation
    must not proceed.

    Expected files per app (per CLAUDE.md §11):
        __init__.py, models.py, serializers.py, views.py, urls.py
    """

    from __future__ import annotations

    import logging
    import os

    from src.msbc.models.schemas.backend_pipeline import ValidationResult

    logger = logging.getLogger(__name__)

    # Files that djcli must produce for each app (CLAUDE.md §11)
    _EXPECTED_APP_FILES: list[str] = [
        "__init__.py",
        "models.py",
        "serializers.py",
        "views.py",
        "urls.py",
    ]


    class ScaffoldValidator:
        """
        Checks that djcli produced the expected file tree for each app.

        Usage::

            validator = ScaffoldValidator()
            result = validator.validate(
                project_path="/tmp/myproject",
                app_names=["users", "products"],
            )
            if not result.success:
                raise RuntimeError(result.errors)
        """

        def validate(
            self,
            project_path: str,
            app_names: list[str],
        ) -> ValidationResult:
            """
            Verify expected scaffold files exist for every app under project_path.

            Parameters
            ----------
            project_path:
                Absolute path to the generated Django project root
                (i.e. os.path.join(output_path, project_name)).
            app_names:
                List of app directory names to check (snake_case, matching djcli output).

            Returns
            -------
            ValidationResult
                success=True  if all expected files exist for every app.
                success=False if any expected file is missing; missing_files lists them
                              in "app_name/filename.py" format.
            """
            missing: list[str] = []

            for app in app_names:
                for fname in _EXPECTED_APP_FILES:
                    full_path = os.path.join(project_path, app, fname)
                    if not os.path.isfile(full_path):
                        rel = f"{app}/{fname}"
                        missing.append(rel)
                        logger.debug("Scaffold missing: %s", full_path)

            success = len(missing) == 0
            errors: list[str] = []
            if not success:
                errors.append(
                    f"djcli scaffold missing {len(missing)} expected file(s): {missing}"
                )
                logger.warning(
                    "Scaffold validation failed for project_path=%s missing=%s",
                    project_path,
                    missing,
                )
            else:
                logger.info(
                    "Scaffold validation passed: project_path=%s apps=%s",
                    project_path,
                    app_names,
                )

            return ValidationResult(
                success=success,
                project_path=project_path,
                missing_files=missing,
                errors=errors,
            )
    ```
